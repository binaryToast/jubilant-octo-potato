// Salt basket drag functionality function createSaltParticles(x, y) { for (let i = 0; i < 8; i++) { const particle = document.createElement('div'); particle.className = 'salt-particle'; const angle = (Math.PI * 2 * i) / 8; const velocity = 3 + Math.random() * 3; const vx = Math.cos(angle) * velocity; const vy = Math.sin(angle) * velocity; particle.style.left = x + 'px'; particle.style.top = y + 'px'; document.body.appendChild(particle); const startTime = Date.now(); const duration = 1500; const animate = () => { const elapsed = Date.now() - startTime; const progress = elapsed / duration; if (progress < 1) { particle.style.left = (x + vx * elapsed / 20) + 'px'; particle.style.top = (y + vy * elapsed / 20 + (elapsed * elapsed) / 4000) + 'px'; particle.style.opacity = (1 - progress) * 0.9; requestAnimationFrame(animate); } else { particle.remove(); } }; animate(); } } function handleBasketMouseDown(e) { isDragging = true; currentBasket = e.currentTarget; currentBasket.style.opacity = '0.7'; } function handleMouseMove(e) { if (!isDragging || !currentBasket) return; createSaltParticles(e.clientX, e.clientY); } function handleMouseUp() { isDragging = false; if (currentBasket) { currentBasket.style.opacity = '0.9'; currentBasket = null; } } saltBasketLeft.addEventListener('mousedown', handleBasketMouseDown); saltBasketRight.addEventListener('mousedown', handleBasketMouseDown); document.addEventListener('mousemove', handleMouseMove); document.addEventListener('mouseup', handleMouseUp); // Banzuke toggle functionality const showBanzukeBtn = document.getElementById('show-banzuke-btn'); const hideBanzukeBtn = document.getElementById('hide-banzuke-btn'); const banzukeContainer = document.getElementById('banzuke-iframe-container'); showBanzukeBtn.addEventListener('click', () => { banzukeContainer.style.display = 'block'; showBanzukeBtn.style.display = 'none'; }); hideBanzukeBtn.addEventListener('click', () => { banzukeContainer.style.display = 'none'; showBanzukeBtn.style.display = 'block'; }); // Load the match data on page load window.addEventListener('load', loadMatchData);
